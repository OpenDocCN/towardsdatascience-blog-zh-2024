        super(MDN, self).__init__()
        self.hidden = nn.Sequential(
            nn.Linear(input_dim, num_hidden),
            nn.Tanh(),
            nn.Linear(num_hidden, num_hidden),
            nn.Tanh(),

        self.z_sigma = nn.Linear(num_hidden, num_mixtures * output_dim)
